--drop table "T_USER"
CREATE TABLE "T_USER"
(
	user_id serial primary key,
	username varchar(40),
	password varchar(40),
	firstname varchar(50),
	lastname varchar(50),
	url varchar(100),
	email varchar(40),
	phone varchar(20)
);

--drop table "ROLE"
CREATE TABLE "ROLE"
(
	role_id serial primary key,
	name varchar(20)
);


CREATE TABLE "SWITCH"
(
	user_id integer references "T_USER" (user_id),
	role_id integer references "ROLE" (role_id)
);
